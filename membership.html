Thank you for being a member!', { htmlBody: htmlBody, attachments: [pdfBlob], name: SENDER_NAME, replyTo: REPLY_TO }); } // ===== MEMBER DATA ===== function getActiveMembers() { var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = ss.getSheetByName(MEMBERSHIP_SHEET_NAME); var data = sheet.getDataRange().getValues(); var headers = data[0].map(function(h){ return String(h).toLowerCase().trim(); }); var firstCol = headers.indexOf('first'); var lastCol = headers.indexOf('last'); var numCol = headers.indexOf('cown #'); var typeCol = headers.indexOf('type'); var statusCol = headers.indexOf('status'); var expCol = headers.indexOf('expiration date'); var emailCol = headers.indexOf('recipient') >= 0 ? headers.indexOf('recipient') : headers.indexOf('email'); var sentCol = headers.indexOf('sent'); var members = []; for (var i = 1; i < data.length; i++) { var first = String(data[i][firstCol] || '').trim(); var last = String(data[i][lastCol] || '').trim(); if (!first && !last) continue; if (String(data[i][statusCol] || '').trim() !== 'Active') continue; var expDate = data[i][expCol]; var expStr = ''; if (expDate instanceof Date) { expStr = (expDate.getMonth()+1) + '/' + expDate.getDate() + '/' + expDate.getFullYear(); } else if (expDate) { expStr = String(expDate).split('T')[0]; } var sentVal = sentCol >= 0 ? } return members; } function findMember(cownNum) { var members = getActiveMembers(); for (var i = 0; i < members.length; i++) { if (members[i].cownNum === cownNum) return members[i]; } return null; } function logSend(member) { try { var ss = SpreadsheetApp.openById(SHEET_ID); break; } } if (sentCol >= 0 && member.row) { var now = new Date(); var dateStr = (now.getMonth()+1) + '/' + now.getDate() + '/' + now.getFullYear(); sheet.getRange(member.row, sentCol + 1).setValue('sent ' + dateStr); } } catch(e) { console.log('Log error:', e); } } // ===== TEST ===== function testSendOne() { var result = sendSingleCard(300); // change to any CoWN # Logger.log(result); } function doGet(e) { var params = e.parameter || {}; return { success: true, member: member.fullName }; }
